🔹 Project: Smart Parking System

A program to manage parking slots using OOP in C++.

#include <iostream>
using namespace std;

class Parking {
  int slots;
public:
  Parking(int s) { slots = s; }
  void park() {
    if(slots > 0) {
      slots--;
      cout << "Car parked. Slots left: " << slots << endl;
    } else {
      cout << "Parking full!" << endl;
    }
  }
};

int main() {
  Parking lot(3);
  lot.park();
  lot.park();
  lot.park();
  lot.park();
  return 0;
}

🔹 Project: bMoney Calculator

Overloaded operators for money handling.

// Class using operator overloading for money
#include <iostream>
using namespace std;

class bMoney {
  double amount;
public:
  bMoney(double a = 0) : amount(a) {}
  bMoney operator+(bMoney m) {
    return bMoney(amount + m.amount);
  }
  void show() { cout << "Amount: Rs. " << amount << endl; }
};

int main() {
  bMoney m1(1500), m2(2000), m3;
  m3 = m1 + m2;
  m3.show();
  return 0;
}
