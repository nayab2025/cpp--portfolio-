🚗 Project: Smart Parking System

A program to manage parking slots using OOP in C++.

#include <iostream>
using namespace std;

class Parking {
  int slots;
public:
  Parking(int s) { slots = s; }
  void park() {
    if(slots > 0) {
      slots--;
      cout << "Car parked. Slots left: " << slots << endl;
    } else {
      cout << "Parking full!" << endl;
    }
  }
};

int main() {
  Parking lot(3);
  lot.park();
  lot.park();
  lot.park();
  lot.park();
  return 0;
}

🔹 Project: bMoney Calculator

Overloaded operators for money handling in C++.

#include <iostream>
using namespace std;

class bMoney {
  double amount;
public:
  bMoney(double a = 0) : amount(a) {}
  bMoney operator+(bMoney m) {
    return bMoney(amount + m.amount);
  }
  void show() { cout << "Amount: Rs. " << amount << endl; }
};

int main() {
  bMoney m1(1500), m2(2000), m3;
  m3 = m1 + m2;
  m3.show();
  return 0;
}

🔹 Project: Simple Calculator

Performs basic arithmetic operations using switch-case in C++.

#include <iostream>
using namespace std;

int main() {
  double num1, num2;
  char op;

  cout << "Enter first number: ";
  cin >> num1;
  cout << "Enter operator (+, -, *, /): ";
  cin >> op;
  cout << "Enter second number: ";
  cin >> num2;

  switch (op) {
    case '+': cout << "Result: " << num1 + num2 << endl; break;
    case '-': cout << "Result: " << num1 - num2 << endl; break;
    case '*': cout << "Result: " << num1 * num2 << endl; break;
    case '/':
      if (num2 != 0)
        cout << "Result: " << num1 / num2 << endl;
      else
        cout << "Error: Division by zero!" << endl;
      break;
    default: cout << "Invalid operator!" << endl;
  }

  return 0;
}

🔹 Project: Factorial Finder

Calculates the factorial of a number using loops.

#include <iostream>
using namespace std;

int main() {
  int num;
  unsigned long long factorial = 1;

  cout << "Enter a positive number: ";
  cin >> num;

  if (num < 0)
    cout << "Error! Factorial of a negative number doesn't exist.";
  else {
    for (int i = 1; i <= num; ++i) {
      factorial *= i;
    }
    cout << "Factorial of " << num << " = " << factorial;
  }

  return 0;
}

🎯 Project: Number Guessing Game

Fun game where the user tries to guess a randomly generated number between 1 and 100.

#include <iostream>
#include <cstdlib>
#include <ctime>
using namespace std;

int main() {
  srand(time(0));
  int secret = rand() % 100 + 1;
  int guess, tries = 0;

  cout << "🎯 Welcome to the Number Guessing Game!" << endl;

  do {
    cout << "Guess a number between 1 and 100: ";
    cin >> guess;
    tries++;

    if (guess < secret)
      cout << "Too low! Try again.\n";
    else if (guess > secret)
      cout << "Too high! Try again.\n";
    else
      cout << "🎉 Correct! You guessed it in " << tries << " tries.\n";

  } while (guess != secret);

  return 0;
}

🔐 Project: User Login System with Files

Implements user registration and login using file I/O in C++.

#include <iostream>
#include <fstream>
#include <string>
using namespace std;

bool login(string user, string pass) {
  ifstream file("users.txt");
  string u, p;
  while (file >> u >> p) {
    if (u == user && p == pass)
      return true;
  }
  return false;
}

int main() {
  int choice;
  cout << "1. Register\n2. Login\nChoice: ";
  cin >> choice;

  string user, pass;
  cout << "Username: ";
  cin >> user;
  cout << "Password: ";
  cin >> pass;

  if (choice == 1) {
    ofstream file("users.txt", ios::app);
    file << user << " " << pass << endl;
    cout << "Registered successfully!";
  } else {
    if (login(user, pass))
      cout << "Login successful!";
    else
      cout << "Invalid credentials!";
  }
  return 0;
}

🧮 Project: Matrix Class with Operator Overloading

Adds two matrices using operator overloading.

#include <iostream>
using namespace std;

class Matrix {
  int a[2][2];
public:
  void input() {
    cout << "Enter 4 elements:\n";
    for (int i=0;i<2;i++)
      for (int j=0;j<2;j++)
        cin >> a[i][j];
  }
  Matrix operator+(Matrix m) {
    Matrix temp;
    for (int i=0;i<2;i++)
      for (int j=0;j<2;j++)
        temp.a[i][j] = a[i][j] + m.a[i][j];
    return temp;
  }
  void display() {
    for (int i=0;i<2;i++) {
      for (int j=0;j<2;j++)
        cout << a[i][j] << " ";
      cout << endl;
    }
  }
};

int main() {
  Matrix m1, m2, m3;
  m1.input(); m2.input();
  m3 = m1 + m2;
  cout << "Result:\n"; m3.display();
  return 0;
}
